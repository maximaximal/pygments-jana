suche(↓liste[1:n] ↓int länge ↓x ↑int i) {
    int j
    j = länge
    while (j>0 && liste[j]!=x) {
	j= j-1
    }
    i = j
    // 1≤i≤länge && x==liste[i] || i=0 && x ist nicht in liste enthalten
}

removeBlanks(↓char[1:n] source ↓int n ↑char[1:m] target ↑ int m) {
    m = 0
    for (int i= 1..n) {
	if (i==1 || source[i] != ' ' || source[i-1] != ' ') {
	    m = m+1
	    target[m] = source[i]
	}
    }
}
